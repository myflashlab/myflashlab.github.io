Packagecom.myflashlab.air.extensions.bluetooth
Classpublic class Bluetooth
InheritanceBluetooth flash.events.EventDispatcher

Bluetooth Extension makes dreams come true for Adobe AIR app developers to transfer data between devices, it lets you connect to other devices through bluetooth and send or receive informations. With some simple methods you are able to turn on bluetooth, make bluetooth device visible, send / receive Strings. Copyright 2015, MyFlashLab. All rights reserved. More Extensions on http://www.MyFlashLabs.com/

Download .ane from: http://www.myflashlabs.com/product/bluetooth-ane-adobe-air-native-extension/

NOTICE: Always make sure to call the dispose method when you are finished with bluetooth.

randomUUIDproperty 
randomUUID:String  [read-only]

every bluetooth connection needs a kind of channel to be able to transfer data back and forth. this channel is called UUID. you may use this property to generate a random UUID and use the generated UUID String in your entire app.

NOTE: do NOT generate a UUID each time your app runs! just do it once and then use the hard-coded string through the whole app. if the UUID of two apps do not match, they will never be able transfer data between themselves.

Examples
The following example shows you how to initialize the Extension in your project and how to use it. 
     import com.myflashlab.air.extensions.bluetooth.Bluetooth;
     import com.myflashlab.air.extensions.bluetooth.BluetoothEvent;
     
     // initialize the extension
     var _ex:Bluetooth = new Bluetooth();
     
     // add the required listeners
     _ex.addEventListener(BluetoothEvent.BLUETOOTH_STATE , onBluetoothState); // dispatches the state of bluetooth whenever it changes
     _ex.addEventListener(BluetoothEvent.COMMUNICATION_STATUS , onCommunicationState); // dispatches the communication state of two devices
     _ex.addEventListener(BluetoothEvent.CONNECTION , onConnectionState); // dispatches the connection state of two devices
     _ex.addEventListener(BluetoothEvent.DIALOG_STATUS , onDialogState); // dispatches the 'enable' and 'visibility' dialog states
     _ex.addEventListener(BluetoothEvent.DISCOVERING_STATUS , onDiscoveringState); // dispatches the device discovering state
     _ex.addEventListener(BluetoothEvent.NEW_DISCOVERD , onNewDiscoverd); // dispatches when new devices are discovered
     _ex.addEventListener(BluetoothEvent.READ_MESSAGE , onMessageReceived); // dispatches whenever a new String message is received from the second device
     _ex.addEventListener(BluetoothEvent.SCAN_MODE , onScanMode); // dispatches to notify you about the scan mode of the device
     _ex.addEventListener(BluetoothEvent.SEND_MESSAGE , onSendMessage); // dispatches when a String message is sent to the second device
     
     // check if bluetooth extension is supported or not
     trace("isSupport: " + _ex.isSupport);
     
     // check if bluetooth is enabled already or not
     trace("isEnable: " + _ex.isEnable);
     
     // use this method to enable bluetooth hardware by asking permissions from user.
     _ex.enable();
     
     // use this method to force enable the Bluetooth without notifying the user.
     _ex.forceEnable();
     
     // use this method to disable the bluetooth
     _ex.disable();
     
     // check if bluetooth is visible to other devices or not
     trace("isVisible: " + _ex.isVisible);
     
     // use this method to make the device visible to other devices. input between 0 and 3600
     // if set to 0 it means that it will remain visible all the time
     _ex.visible(160);
     
     // use this method to make the device invisible
     _ex.inVisisble();
     
     // use this property to get an array list of objects from already paired devices
     var arr:Array = _ex.pairedList;
     var obj:Object;
     var i:int;
     for (i = 0; i < arr.length; i++) 
     {
         obj = arr[i];
         trace("paired >>" + obj.device + " : " + obj.mac);
     }
     
     // use this method to start discovering for other bluetoothes
     _ex.startDiscovery();
     
     // use this method to stop the discovery process
     _ex.stopDiscovery();
     
     // check if the device is in discovery process or not
     trace("isDiscovering:" + _ex.isDiscovery);
     
     // use this method to pair the device with another one by passing the other device's mac address
     _ex.pairWith("C4:88:E5:5B:AC:6D");
     
     // use this method to unpair the device from another one by specifying the other device's mac address
     _ex.unPairWith("C4:88:E5:5B:AC:6D");
     
     // every bluetooth connection needs a kind of channel to be able to transfer data back and forth. this channel is called UUID
     // you may use this property to generate a random UUID and use the generated UUID String in your entire app.
     // NOTE: do NOT generate a UUID each time your app runs! just do it once and then use the hard-coded string through the whole app
     // if the UUID of two apps do not match, they will never be able transfer data between themselves.
     trace("getRandomUUID >>" +  _ex.randomUUID);
     
     // use this method to initialize a communication between the two devices.
     // NOTE: it's totally find if you forget about generating a UUID!
     _ex.initCommunicationService();
     
     // After initializing the communication between two devices, use this method to connect to the target device via its mac address
     _ex.connectTo("C4:88:E5:5B:AC:6D");
     
     // besides the event listeners, you may use this property to know about the connection status of the devices
     trace("connectionStatus :" + _ex.connectionStatus);
     
     // After the devices are connected, you may use this method to send a string message out to the other device
     _ex.sendMessage("Hello second device!");
     
     // call this method to disconnect two devices from each other
     _ex.disconnect();
     
     function onBluetoothState(e:BluetoothEvent):void
     {
         trace("state >> " + e.param);
     }
     
     function onCommunicationState(e:BluetoothEvent):void
     {
         trace("communication >> " + e.param);
     }
     
     function onConnectionState(e:BluetoothEvent):void
     {
         trace("connection >> " + e.param);
     }
     
     function onDialogState(e:BluetoothEvent):void
     {
         trace("dialogStatus >> " + e.param);
     }
     
     function onDiscoveringState(e:BluetoothEvent):void
     {
         trace("discovering >> " + e.param);
     }
     
     function onNewDiscoverd(e:BluetoothEvent):void
     {
         var obj:Object = e.param;
         trace("newDiscoverd >> " + "device = " + obj.device + " mac :" + obj.mac);
     }
     
     function onMessageReceived(e:BluetoothEvent):void
     {
         trace("readMessage >> " + e.param);
     }
     
     function onScanMode(e:BluetoothEvent):void
     {
         trace("scanMode >> " + e.param);
     }
     
     function onSendMessage(e:BluetoothEvent):void
     {
         // when sending a message out to the other device, this listener will be called and on its param will return the same string message you had sent
         // but if for any reason, the message is not sent, the param will be "unableToSend"! this way, you will know if the message was not sent successfully
         trace("sendMessage >> " + e.param);
     }
